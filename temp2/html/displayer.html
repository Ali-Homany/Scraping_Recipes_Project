def create_steps(recipe):
    '''
    Returns:
        - (list) list of steps of the given recipe
    '''
    result = '<ol>'
    for step in recipe['steps']:
        result += f'<li>{step}</li>'
    result += '</ol>'
    return result


def create_ingredients(recipe):
    '''
    Returns:
        - (list) list of ingredients of the given recipe
    '''
    result = '<ul>'
    for ingredient in recipe['ingredients']:
        result += f'<li>{ingredient}</li>'
    result += '</ul>'
    return result

    '''
    Returns:
